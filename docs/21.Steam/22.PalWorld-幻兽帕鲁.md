## PalWorld-幻兽帕鲁

### 官方开服WIKI

https://docs.palworldgame.com/

### steamcmd命令

```cmd
"steamcmd.exe" +login anonymous +app_update 2394010 validate +quit
```

### 配置翻译（联动前）

```xml
	Difficulty=None; 设置游戏难度
    DayTimeSpeedRate=1.000000, ; 设置白天时间流逝的速率
    NightTimeSpeedRate=1.000000, ; 设置夜晚时间流逝的速率
    ExpRate=1.000000, ; 设置经验获取速率
    PalCaptureRate=1.000000, ; 设置捕捉Pal（游戏中的生物）的几率

    PalDamageRateDefense=1.000000, ; 设置Pal的防御伤害比率

    PalStomachDecreaceRate=1.000000, ; 设置Pal饱食度下降速率
    PalStaminaDecreaceRate=1.000000, ; 设置Pal耐力下降速率
    PalAutoHPRegeneRate=1.000000, ; 设置Pal的自动生命恢复速率
    PalAutoHpRegeneRateInSleep=1.000000, ; 设置Pal在睡眠中的生命恢复速率
    BuildObjectDamageRate=1.000000, ; 设置建筑物受到的伤害速率
    BuildObjectDeteriorationDamageRate=1.000000, ; 设置建筑物的恶化（老化）伤害速率
    CollectionDropRate=1.000000, ; 设置收集物品掉落速率
    CollectionObjectHpRate=1.000000, ; 设置收集物品耐久度比率
    CollectionObjectRespawnSpeedRate=1.000000, ; 设置收集物品重生速度比率
    EnemyDropItemRate=1.000000, ; 设置敌人掉落物品的比率
    DeathPenalty=All, ; 设置死亡惩罚。None 是关闭死亡掉落，Item只掉物品不掉装备，ItemAndEquipment掉物品和装备，All全都掉 
    bEnablePlayerToPlayerDamage=False, ; 是否允许玩家之间造成伤害
    bEnableFriendlyFire=False, ; 是否允许友军火力（误伤）
    bEnableInvaderEnemy=True, ; 是否启用入侵者敌人
    bActiveUNKO=False, ; 未知选项UNKO
    bEnableAimAssistPad=True, ; 是否为手柄启用瞄准辅助
    bEnableAimAssistKeyboard=False, ; 是否为键盘启用瞄准辅助
    DropItemMaxNum=3000, ; 设置掉落物品的最大数量
    DropItemMaxNum_UNKO=100, ; 未知选项UNKO的数量设置
    BaseCampMaxNum=128, ; 设置基地营地的最大数量
    BaseCampWorkerMaxNum=15, ; 设置基地营地工人的最大数量
    DropItemAliveMaxHours=1.000000, ; 设置掉落物品在世界中存活的最大时间（小时）
    bAutoResetGuildNoOnlinePlayers=False, ; 未知行会选项
    AutoResetGuildTimeNoOnlinePlayers=72.000000, ; 玩家不在线时行会自动重置时间（小时）
    GuildPlayerMaxNum=20, ; 行会玩家最大数量
    PalEggDefaultHatchingTime=72.000000, ; Pal蛋默认孵化时间（小时）
    WorkSpeedRate=1.000000, ; 工作速率
    bIsMultiplay=False, ; 是否为多人游戏
    bIsPvP=False, ; 是否开启玩家对战
    bCanPickupOtherGuildDeathPenaltyDrop=False, ; 是否可以拾取其他行会因死亡惩罚掉落物
    bEnableNonLoginPenalty=True, ; 是否启用未登录惩罚
    bEnableFastTravel=True, ; 是否启用快速旅行
    bIsStartLocationSelectByMap=True, ; 是否允许通过地图选择起始位置
    bExistPlayerAfterLogout=False, ; 玩家登出后是否仍然存在于游戏世界中
    bEnableDefenseOtherGuildPlayer=False, ; 是否允许防御其他行会玩家
    CoopPlayerMaxNum=4, ; 合作模式玩家最大数量
    ServerPlayerMaxNum=32, ; 服务器玩家最大数量
    ServerName="脱脱测试服", ; 服务器名称
    ServerDescription="", ; 服务器描述
    AdminPassword="", ; 管理员密码
    ServerPassword="123123", ; 服务器密码
    PublicPort=8211, ; 服务器公开端口
    PublicIP="", ; 服务器公开IP地址
    RCONEnabled=False, ; 是否启用RCON（远程控制）
    RCONPort=25575, ; RCON端口
    Region="", ; 服务器区域
    bUseAuth=True, ; 是否使用认证
    BanListURL="https://api.palworldgame.com/api/banlist.txt" ; 封禁列表网址
```
