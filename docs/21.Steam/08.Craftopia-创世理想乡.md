## Craftopia-创世理想乡

### steamcmd

```cmd
steamcmd +login anonymous +app_update 1670340 validate +quit
```

### 启动

```cmd
D:\steamcmd\steamapps\common\CraftopiaDedicatedServer\Craftopia.exe
```

### 配置文件

ServerSetting.ini

```xml

; Game World Data Setting - name, difficulty, gameMode
; name : 世界名城
; difficulty ： 游戏难度 0- 简单, 1- 普通(默认难度), 2- 困难, 3- 非常难
; gameMode ： 游戏模式 1- 正常世界(默认), 2- 自由建造, 3- 创意模式

; Server Setting - port : used when not using steam lobby function
; maxPlayerNumber : 玩家人数最大1-127
; usePassword : 使用服务器密码(1- 开启 / 0- 关闭)
; serverPassword : 服务器密码8位数

; Graphics Setting - vSyncCount, maxFPS, grassBend, clothSimOption
; vSyncCount : 垂直同步, 0- 关闭, 1- 1次1帧数  2- 2次1帧数
; maxFPS : 最大FPS值,-1不限制
; grassBend : 草地弯曲(1- 启用 / 0- 关闭)
; ClothSimOption : 布料模拟，1-设置个人的动画，2-设置所有人的动画

; Save Setting - autoSaveSec, autoSavePerHour
; autoSaveSec : 自动保存时间:秒
; autoSavePerHour : 启用每小时自动保存（1-True/0-False）
; savePath : 存档保存路径

; Creative Mode Setting -
; quickCraft : 快速建造:不需要材料,1启动,2关闭
; ageLevel : 游戏时代的等级，影响效果制作配方
; islandLevel : 岛屿等级控制，影响怪物和副本级别，设置为-1禁用
; noDeath : 玩家角色不会死亡（1-开启/0-关闭)
; noDamage : 玩家角色不会受到伤害（1-开启/0-关闭）
; noHunger : 玩家角色不会饥饿（1-开启/0-关闭）
; infinitStamina :玩家角色具有无穷的耐力（1-开启/0-关闭）
; forceDayTime :游戏世界日时间覆盖，0到24，设置为-1禁用
; buildingIgnoreDamage : 建筑物不会受到损害（1-开启/0-关闭）
; noBuild : 玩家无法放置新建筑（1-开启/0-关闭）

; Creative Mode Player Status Override - apply at gameStart
; Level : override character level, set to 0 to disable
; Health : override character health, set to 0 to disable
; Mana : override character mana, set to 0 to disable
; Stamina : override character stamia, set to 0 to disable
; Money : override character money
; SkillPoint : override character skill point
; EnchantPoint : override character enchant point

[GameWorld]
name=XiongTianQi
difficulty=3
gameMode=1

[Host]
port=6587
maxPlayerNumber=127
usePassword=0
serverPassword=00000000
bindAddress=0.0.0.0

[Graphics]
vSyncCount=0
maxFPS=60
grassBend=0
clothSimOption=2

[Save]
autoSaveSec=300
autoSavePerHour=1
savePath=DedicatedServerSave/

[CreativeModeSetting]
quickCraft=1
ageLevel=7
islandLevel=-1
noDeath=1
noDamage=1
noHunger=1
infinitStamina=1
forceDayTime=-1
buildingIgnoreDamage=0
noBuild=0

[CreativeModePlStatus]
Level=0
Health=0
Mana=0
Stamina=0
Money=1000
SkillPoint=0
EnchantPoint=0

```
