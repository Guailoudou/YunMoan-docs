import os
import pandas as pd

def count_nicknames_and_aggregate_amounts(folder_path):
    # 用于存储昵称出现的次数和金额
    nickname_data = {}
    # 抽象弹幕关键词
    barrage_data=['主人','踩我','骂我','爽了','好爽','踢我','爽','抽我','打我','撅我','进入我','老婆']
    # 遍历目标文件夹中的所有文件
    for filename in os.listdir(folder_path):
        if filename.endswith('.csv'):
            file_path = os.path.join(folder_path, filename)
            print(file_path)
            # 读取CSV文件，跳过第一行
            df = pd.read_csv(file_path, skiprows=1)
            # 检查第二列是否存在
            if df.shape[1] > 1:
                for index, row in df.iterrows():
                    nickname = row.iloc[1]  # 名称
                    barrage = str(row.iloc[3]) #弹幕
                    if isinstance(nickname, str):
                        # 初始化昵称数据
                        if nickname not in nickname_data:
                            nickname_data[nickname] = {
                                '弹幕数量': 0,
                                'SC金额': 0,
                                '舰队金额': 0,
                                '礼物金额': 0,
                                '抽象次数': 0,
                                '抽象弹幕': []
                            }

                        # 针对文件名包含“弹幕”，统计弹幕数量
                        if '弹幕' in filename:
                            nickname_data[nickname]['弹幕数量'] += 1
                        if any(keyWord in barrage for keyWord in barrage_data):
                            nickname_data[nickname]['抽象次数'] +=1
                            nickname_data[nickname]['抽象弹幕'].append(barrage)
                        # 针对不同文件名聚合金额
                        if 'SC' in filename and df.shape[1] > 3:
                            sc_amount = row.iloc[3]  # 第四列
                            if isinstance(sc_amount, (int, float)):
                                nickname_data[nickname]['SC金额'] += sc_amount
                        if '舰队' in filename and df.shape[1] > 5:
                            fleet_amount = row.iloc[5]  # 第六列
                            if isinstance(fleet_amount, (int, float)):
                                nickname_data[nickname]['舰队金额'] += fleet_amount/1000
                        if '礼物' in filename and df.shape[1] > 5:
                            gift_amount = row.iloc[5]  # 第六列
                            gift_count = row.iloc[4]
                            if isinstance(gift_amount, (int, float)):
                                nickname_data[nickname]['礼物金额'] += gift_count*gift_amount/1000

    return nickname_data

def write_counts_to_excel(data, output_file):
    # 将字典转换为DataFrame
    df = pd.DataFrame.from_dict(data, orient='index')
    df.reset_index(inplace=True)
    df.columns = ['昵称', '弹幕数量', 'SC金额', '舰队金额', '礼物金额','抽象次数','抽象弹幕']
    
    # 计算金额合计
    df['合计金额'] = df['SC金额'] + df['舰队金额'] + df['礼物金额']
    
    # 输出到Excel文件
    df.to_excel(output_file, index=False)
